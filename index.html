function drawShagai(ctx, x, y, radius, side) { ctx.save(); ctx.translate(x, y); // 1. Металл алтлаг эффект гаргах Gradient let goldGrad = ctx.createLinearGradient(-radius, -radius, radius, radius); goldGrad.addColorStop(0, "#D4AF37"); // Dark Gold goldGrad.addColorStop(0.2, "#FFDF00"); // Bright Gold goldGrad.addColorStop(0.5, "#FFFFFF"); // Reflection (Highlight) goldGrad.addColorStop(0.8, "#CFB53B"); // Old Gold goldGrad.addColorStop(1, "#8B6508"); // Shadow Gold ctx.fillStyle = goldGrad; ctx.strokeStyle = "#8B6508"; ctx.lineWidth = 1; // Сүүдэр нэмэх ctx.shadowColor = "rgba(0,0,0,0.4)"; ctx.shadowBlur = 12; ctx.shadowOffsetY = 6; // 2. Дүрс бүрийн хэлбэрийг тусгайлан зурах ctx.beginPath(); switch(side.name) { case "Морь": // image_d53cd6.jpg-ийн зүүн дээд талынх шиг төвгөр ctx.moveTo(-radius, 0); ctx.bezierCurveTo(-radius, -radius, radius, -radius, radius, 0); ctx.bezierCurveTo(radius, radius*0.5, 0, radius, -radius, 0); break; case "Тэмээ": // Баруун дээд талынх шиг хоёр бөхтэй мэт хотгор ctx.moveTo(-radius, -radius*0.2); ctx.quadraticCurveTo(-radius*0.5, -radius*1.2, 0, -radius*0.5); ctx.quadraticCurveTo(radius*0.5, -radius*1.2, radius, -radius*0.2); ctx.lineTo(radius, radius*0.5); ctx.quadraticCurveTo(0, radius*1.2, -radius, radius*0.5); ctx.closePath(); break; case "Хонь": // Зүүн доод талынх шиг бөөрөнхий ctx.ellipse(0, 0, radius, radius * 0.8, 0, 0, Math.PI * 2); break; case "Ямаа": // Баруун доод талынх шиг нарийн, шовх талтай ctx.moveTo(-radius, 0); ctx.lineTo(-radius*0.5, -radius*0.8); ctx.lineTo(radius*0.5, -radius*0.8); ctx.lineTo(radius, 0); ctx.lineTo(0, radius*0.8); ctx.closePath(); break; } ctx.fill(); ctx.stroke(); // 3. Гэрэлтэлтийн эффект (Гялбаа нэмэх) ctx.globalAlpha = 0.3; ctx.fillStyle = "white"; ctx.beginPath(); ctx.ellipse(-radius*0.3, -radius*0.3, radius*0.2, radius*0.1, Math.PI/4, 0, Math.PI*2); ctx.fill(); ctx.restore(); }
