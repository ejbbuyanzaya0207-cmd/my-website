function drawShagai(ctx, x, y, radius, side) { ctx.save(); ctx.translate(x, y); // 1. Металл алтлаг эффект гаргах Gradient // Зураг дээрх шиг гэрэл ойж буйг харуулахын тулд цагаан болон алтлаг өнгийг сүлжсэн let goldGrad = ctx.createLinearGradient(-radius, -radius, radius, radius); goldGrad.addColorStop(0, "#8B6508"); // Бараан алтлаг (сүүдэр) goldGrad.addColorStop(0.2, "#D4AF37"); // Үндсэн алтлаг goldGrad.addColorStop(0.5, "#FFFFFF"); // Гэрэл туссан хэсэг (Цагаан гялбаа) goldGrad.addColorStop(0.7, "#FFDF00"); // Тод алтлаг goldGrad.addColorStop(1, "#B8860B"); // Гүн алтлаг ctx.fillStyle = goldGrad; ctx.strokeStyle = "rgba(139, 101, 8, 0.5)"; // Нарийн хүрээ ctx.lineWidth = 1; // Сүүдэр нэмэх (Шагайг хөндий харагдуулна) ctx.shadowColor = "rgba(0,0,0,0.4)"; ctx.shadowBlur = 10; ctx.shadowOffsetY = 5; // 2. Дүрс бүрийн хэлбэрийг тусгайлан зурах (Bezier Curves) ctx.beginPath(); switch(side.name) { case "Морь": // Нуруу хэсэг нь төвгөр, бага зэрэг муруй хэлбэр ctx.moveTo(-radius, 0); ctx.bezierCurveTo(-radius, -radius * 1.2, radius, -radius * 1.2, radius, 0); ctx.bezierCurveTo(radius, radius * 0.8, -radius, radius * 0.8, -radius, 0); break; case "Тэмээ": // Хоёр бөхтэй мэт хотгор хэлбэр ctx.moveTo(-radius, -radius * 0.2); ctx.quadraticCurveTo(-radius * 0.5, -radius * 1.3, 0, -radius * 0.5); ctx.quadraticCurveTo(radius * 0.5, -radius * 1.3, radius, -radius * 0.2); ctx.lineTo(radius, radius * 0.5); ctx.quadraticCurveTo(0, radius * 1.3, -radius, radius * 0.5); break; case "Хонь": // Илүү бөөрөнхий, дүүрэн хэлбэр ctx.ellipse(0, 0, radius, radius * 0.85, 0, 0, Math.PI * 2); break; case "Ямаа": // Нарийн, хажуу талдаа шовх ирмэгтэй хэлбэр ctx.moveTo(-radius, 0); ctx.lineTo(-radius * 0.6, -radius * 0.9); ctx.lineTo(radius * 0.6, -radius * 0.9); ctx.lineTo(radius, 0); ctx.quadraticCurveTo(0, radius * 1.2, -radius, 0); break; } ctx.fill(); ctx.stroke(); // 3. Зураг дээрх шиг нэмэлт "Гялбаа" (Highlight) нэмэх ctx.globalAlpha = 0.4; ctx.fillStyle = "white"; ctx.beginPath(); // Шагайн зүүн дээд талд жижиг зууван цагаан гялбаа нэмнэ ctx.ellipse(-radius * 0.4, -radius * 0.4, radius * 0.3, radius * 0.15, Math.PI / 4, 0, Math.PI * 2); ctx.fill(); ctx.restore(); }
